<!DOCTYPE html>

<html>

<head>

    <title>CSV Image Grid Tool</title>

    <script src="https://unpkg.com/react@18/umd/react.production.min.js"></script>

    <script src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>

    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>

    <script src="https://unpkg.com/html2canvas@1.4.1/dist/html2canvas.min.js"></script>

    <style>

        body { font-family: sans-serif; background: #f0f2f5; margin: 0; padding: 20px; }

        .grid { display: grid; grid-template-columns: repeat(auto-fill, 150px); gap: 10px; justify-content: center; background: white; padding: 20px; border-radius: 8px; }

        .progress-container { width: 100%; height: 10px; background: #ddd; border-radius: 5px; margin: 20px 0; overflow: hidden; }

        .progress-bar { height: 100%; background: #4caf50; transition: width 0.3s; }

    </style>

</head>

<body>

    <div id="root"></div>


    <script type="text/babel">

        const { useState, useRef } = React;


        function App() {

            const [urls, setUrls] = useState([]);

            const [progress, setProgress] = useState(0);

            const gridRef = useRef();


            const handleFile = (e) => {

                const reader = new FileReader();

                reader.onload = (ev) => {

                    const links = ev.target.result.split(/\n/).map(l => l.split(',')[0].trim()).filter(l => l.startsWith('http'));

                    setUrls(links);

                };

                reader.readAsText(e.target.files[0]);

            };


            const download = async () => {

                setProgress(20);

                const canvas = await html2canvas(gridRef.current, { useCORS: true, scale: 2 });

                setProgress(80);

                const link = document.createElement('a');

                link.download = 'grid.jpg';

                link.href = canvas.toDataURL('image/jpeg', 0.9);

                link.click();

                setProgress(100);

            };


            return (

                <div style={{maxWidth: '1200px', margin: 'auto'}}>

                    <h1>CSV Image Grid</h1>

                    <input type="file" accept=".csv" onChange={handleFile} />

                    <button onClick={download} disabled={!urls.length} style={{marginLeft: '10px', padding: '8px 16px'}}>Download JPG</button>

                    

                    {progress > 0 && <div className="progress-container"><div className="progress-bar" style={{width: progress + '%'}}></div></div>}

                    

                    <div ref={gridRef} className="grid" style={{marginTop: '20px'}}>

                        {urls.map((u, i) => <img key={i} src={u} crossOrigin="anonymous" style={{width: '150px', height: '150px', objectFit: 'cover'}} />)}

                    </div>

                </div>

            );

        }


        const root = ReactDOM.createRoot(document.getElementById('root'));

        root.render(<App />);

    </script>

</body>

</html>
